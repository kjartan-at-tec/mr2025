\documentclass{standalone}
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage{pgf,tikz}
\usepackage{setspace}
\usepackage{pgfplots}

\begin{document}
\def\TT{1}
\pgfmathsetmacro{\ww}{1/\TT}
\pgfmathsetmacro{\hh}{0.4/\ww}
\pgfmathsetmacro{\zz}{0.5}
\pgfmathsetmacro{\realpart}{-\ww*\zz}
\pgfmathsetmacro{\impart}{\ww*sqrt(1-pow(\zz,2))}
\pgfmathsetmacro{\phshift}{-acos(\zz)}

\begin{tikzpicture}
  \begin{axis}[
    width=12cm,
    height=8cm,
    xlabel={$t$},
    ylabel={$y(t)$},
    xmin={-2.5*\hh},
    xmax={30*\hh},
    grid=both,
    minor x tick num = 3,
    minor y tick num = 3,
    minor grid style={black!30},
    %ytick=\empty,
    xtick={0,2,4,6,8,10,12},
    every major grid/.style={red, opacity=0.5},
    ]
    \addplot+[thick, black, no marks, domain=-2*\hh:30*\hh, samples=100, variable=x] {(x>0)*((1 - exp(\realpart*x) / sqrt(1-\zz * \zz) * sin(deg(\impart*x) + acos(\zz)))};
  \end{axis} 
\end{tikzpicture}
\end{document}
